58 void Link::create_Devices()
59 {
60  if (AvgSpanLength < 0)
61  {
62  return;
63  }
64 
65  Devices.clear();
66  numLineAmplifiers = floor(Length / AvgSpanLength);
67 
68  if (ceil(Length / AvgSpanLength) == numLineAmplifiers)
69  {
70  numLineAmplifiers--;
71  }
72 
73  double SpanLength = Length / (numLineAmplifiers + 1);
74 
75  for (int i = 0; i < numLineAmplifiers; i++)
76  {
77  Devices.push_back(std::shared_ptr<Fiber>(new Fiber(SpanLength)));
78  Devices.push_back(std::shared_ptr<InLineAmplifier>(
79  new InLineAmplifier((Fiber &)*Devices.back())));
80  }
81 
82  //There's an extra fiber segment in the end of the link
83  Devices.push_back(std::shared_ptr<Fiber>(new Fiber(SpanLength)));
84 
85  //There's a preamplifier in the node's entrance.
86  //It compensates the fiber segment loss and also the switching element loss.
87  Devices.push_back(std::shared_ptr<PreAmplifier>(new PreAmplifier((
88  Fiber &)*Devices.back(), *Destination.lock())));
89 }
