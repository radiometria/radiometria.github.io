30 void NSGA2_Generation::evalCrowdingDistances(int ParetoFront)
31 {
32  if (isEvaluated)
33  {
34  return;
35  }
36 
37  auto front = getParetoFront(ParetoFront);
38  unsigned int numParameters = people.front()->getNumParameters();
39 
40  //Initialize crowding distances
41  for (auto &individual : front)
42  {
43  individual->crowdingDistance = 0;
44  }
45 
46  for (unsigned int par = 0; par < numParameters; par++)
47  {
48  struct Param
49  {
50  Param(std::shared_ptr<NSGA2_Individual> individual, double value) :
51  individual(individual), value(value) {}
52  std::shared_ptr<NSGA2_Individual> individual;
53  double value;
54  bool operator < (const Param &other) const
55  {
56  return value < other.value;
57  }
58  };
59 
60  std::vector<Param> IndivList;
61 
62  for (auto indiv : front)
63  {
64  IndivList.push_back(Param(indiv, indiv->getParameterValue(par)));
65  }
66 
67  std::sort(IndivList.begin(), IndivList.end());
68 
69  double minValue = IndivList.front().value;
70  double maxValue = IndivList.back().value;
71 
72  //Extreme values have infinite crowding distance
73  IndivList.front().individual->crowdingDistance =
74  std::numeric_limits<double>::max();
75  IndivList.back().individual->crowdingDistance =
76  std::numeric_limits<double>::max();
77 
78  for (size_t i = 1; i < IndivList.size() - 1; i++)
79  {
80  if (IndivList[i].individual->crowdingDistance ==
81  std::numeric_limits<double>::max())
82  {
83  continue;
84  }
85 
86  IndivList[i].individual->crowdingDistance +=
87  (IndivList[i + 1].value - IndivList[i - 1].value) / (maxValue - minValue);
88  }

107 void NSGA2_Generation::evalParetoFront()
108 {
109  if (isEvaluated)
110  {
111  return;
112  }
113 
114  int numNotInParetoFront = 0;
115  int currentParetoFront = 1;
116 
117  for (auto &indiv : people)
118  {
119  indiv->crowdingDistance = indiv->paretoFront = -1;
120  }
121 
122  do
123  {
124  std::vector<std::shared_ptr<NSGA2_Individual>> currentFront;
125  numNotInParetoFront = 0;
126 
127  for (auto &indiv : people)
128  {
129  if (indiv->paretoFront != -1)
130  {
131  continue;
132  }
133 
134  bool isDominated = false;
135 
136  for (auto &other : people)
137  {
138  if ((other->paretoFront != -1) || isDominated)
139  {
140  continue;
141  }
142  isDominated = isDominated || (indiv->isDominatedBy(other));
143  }
144 
145  if (!isDominated)
146  {
147  currentFront.push_back(indiv);
148  continue;
149  }
150 
151  numNotInParetoFront++;
152  }
153 
154  for (auto &indiv : currentFront)
155  {
156  indiv->paretoFront = currentParetoFront;
157  }
158 
159  currentParetoFront++;
160  }
161  while (numNotInParetoFront != 0);

280 {
281  std::uniform_int_distribution<> dist(0, people.size() - 1);
282 
283  //selects random Individual
284  auto individual = people.begin() + dist(random_generator);
285 
286  //selects binaryTournamentParameter other individuals to "duel" with
287  //individual, in terms of Pareto Front.
288  for (unsigned selec = 0; selec < NSGA2::binaryTournamentParameter; selec++)
289  {
290  auto adversary = people.begin() + dist(random_generator);
291 
292  while (adversary == individual)
293  {
294  adversary = people.begin() + dist(random_generator);
295  }
296

298  {
299  individual = adversary;
