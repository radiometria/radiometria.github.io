176 void Simulation_StatisticalTrend::create_Simulations()
177 {

180  place_Regenerators(T);
181  }
182 
183  for (int i = 0; i < NumRepetitions; i++)
184  {
185  //Creates a copy of the topology.
186  std::shared_ptr<Topology> TopologyCopy(new Topology(*T));
187 
188  //Creates the RWA Algorithms
189  std::shared_ptr<RoutingAlgorithm> R_Alg =
190  RoutingAlgorithm::create_RoutingAlgorithm(Routing_Algorithm, TopologyCopy);
191  std::shared_ptr<WA::WavelengthAssignmentAlgorithm> WA_Alg =
192  WA::WavelengthAssignmentAlgorithm::create_WavelengthAssignmentAlgorithm(
193  WavAssign_Algorithm, TopologyCopy);
194  std::shared_ptr<RegeneratorAssignmentAlgorithm> RA_Alg;
195 
196  if (Type == TranslucentNetwork)
197  {
198  RA_Alg = RegeneratorAssignmentAlgorithm::create_RegeneratorAssignmentAlgorithm(
199  RegAssignment_Algorithm, TopologyCopy);
200  }
201  else
202  {
203  RA_Alg = nullptr;
204  }
205 
206  //Creates the Call Generator and the RWA Object
207  auto Generator = std::make_shared<CallGenerator>(TopologyCopy, NetworkLoad);
208  auto RWA = std::make_shared<RoutingWavelengthAssignment>(R_Alg, WA_Alg, RA_Alg,
209  ModulationScheme::DefaultSchemes, TopologyCopy);
210 
211  //Push simulation into stack
212  simulations.push_back(
213  std::make_shared<NetworkSimulation>(Generator, RWA, NumCalls));
214  }
