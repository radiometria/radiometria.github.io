44 void Simulation_TransparencyAnalysis::run()
45 {
46  if (!hasLoaded)
47  {
48  load();
49  }
50 
51  for (double avgSpan = minAvgLinkSpan; avgSpan <= maxAvgLinkSpan;
52  avgSpan += stepAvgLinkSpan)
53  {
54  for (double inOSNR = minOSNR; inOSNR <= maxOSNR; inOSNR += stepOSNR)
55  {
56  Signal::InputOSNR = Gain(inOSNR, Gain::dB);
57 
58  std::shared_ptr<Topology> TopCopy(new Topology(*T));
59  TopCopy->set_avgSpanLength(avgSpan);
60  std::weak_ptr<Node> origin, destination;
61 
62  for (auto node : TopCopy->Nodes)
63  {
64  if (node->ID == Orig_Origin->ID)
65  {
66  origin = node;
67  }
68 
69  if (node->ID == Orig_Destination->ID)
70  {
71  destination = node;
72  }
73  }
74 
75  std::shared_ptr<ShortestPath> SP(new ShortestPath(TopCopy));
76  std::shared_ptr<WA::FirstFit> FF(new WA::FirstFit(TopCopy));
77 
78  RoutingWavelengthAssignment RWA(SP, FF, ModulationScheme::DefaultSchemes,
79  TopCopy);
80  std::shared_ptr<Call> C(new Call(origin, destination, *maxBitrate));
81 
82  if (RWA.routeCall(C) != nullptr)
83  {
84  TransparentPoints.push_back(InLineDistance_OSNR_Point(avgSpan, inOSNR));
85  }
86  else
87  {
88  OpaquePoints.push_back(InLineDistance_OSNR_Point(avgSpan, inOSNR));
89  }
90  }
91  }
92 
93  std::cout << std::endl << "* * RESULTS * *" << std::endl;
94 
95  if (!TransparentPoints.empty())
96  {
97  std::cout << std::endl << "Transparent Points: " << std::endl;
98 
99  for (auto &pair : TransparentPoints)

102  }

359 void Simulation_TransparencyAnalysis::find_OriginDestination()
360 {
361  ShortestPath SP(T);
362  double maxLength = -1;
363 
364  for (auto orig : T->Nodes)
365  {
366  for (auto dest : T->Nodes)
367  {
368  if (orig == dest)
369  {
370  continue;
371  }
372 
373  std::shared_ptr<Call> DummyCall(new Call(orig, dest, 0));
374  auto links = SP.route(DummyCall);
375  double length = 0;
376 
377  for (auto link : links)
378  {
379  length += link.lock()->Length;
380  }
381 
382  if (maxLength < length)
383  {
384  maxLength = length;
385  Orig_Origin = orig;
386  Orig_Destination = dest;
387  }
388  }
389  }
