647 void Simulation_PSROptimization::runPSR()
648 {
649  if (!hasRun)
650  {
651  Fitness::T = T;
652  Fitness::Variant = Variant;
653 
654  switch (Variant)
655  {
656  case PowerSeriesRouting::Variant_MatricialPSR:
657  PSO_Optim =
658  std::shared_ptr<ParticleSwarmOptimization<double, Fitness, Compare>>
659  (new ParticleSwarmOptimization<double, Fitness, Compare>
660  (P, G, std::pow(NMax - NMin + 1, Costs.size()), XMin, XMax, VMin, VMax));
661  break;
662 
663  case PowerSeriesRouting::Variant_TensorialPSR:
664  PSO_Optim =
665  std::shared_ptr<ParticleSwarmOptimization<double, Fitness, Compare>>
666  (new ParticleSwarmOptimization<double, Fitness, Compare>
667  (P, G, Costs.size() * (NMax - NMin + 1), XMin, XMax, VMin, VMax));
668  break;
669 
670  case PowerSeriesRouting::Variant_LocalPSR:
671  PSO_Optim =
672  std::shared_ptr<ParticleSwarmOptimization<double, Fitness, Compare>>
673  (new ParticleSwarmOptimization<double, Fitness, Compare>
674  (P, G, std::pow
675  (NMax - NMin + 1, Costs.size()) * T->Nodes.size() * (T->Nodes.size() - 1),
676  XMin, XMax, VMin, VMax));
677  break;
678 
679  case PowerSeriesRouting::Variant_AWR:
680  PSO_Optim =
681  std::shared_ptr<ParticleSwarmOptimization<double, Fitness, Compare>>
682  (new ParticleSwarmOptimization<double, Fitness, Compare>
683  (P, G, Costs.size() - 1, 0, std::acos(-1), VMin, VMax));
684  break;
685  }
686  }
687 
688  std::cout << std::endl << "* * RESULTS * *" << std::endl;
689 
690  std::ofstream logFile(LogFilename);
691 
692  for (unsigned i = 1; i <= G; i++)
693  {
694  if (!hasRun && (!BestParticle || BestParticle->bestFit != 0.0))
695  {
696  PSO_Optim->run_generation();
697  }
698 
699  BestParticle = PSO_Optim->BestParticle;
